# 有向图的构建 (C++ 标程)

这是为各位同学准备的**有向图**建图标程。与无向图最大的区别在于：读入边 `u v w` 时，**只需建立从 u 到 v 的单向连接**，而不需要反向建立 `v` 到 `u` 的连接。

## 1. 邻接矩阵版本 (Adjacency Matrix)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int INF = 1e9; // 表示无穷大，代表两点之间没有边

int main() {
    int n, m; // n 个顶点，m 条边
    cout << "请输入顶点数 n 和边数 m: ";
    if (!(cin >> n >> m)) return 0;

    // 1. 显式定义 vertex 数组
    vector<int> vertex(n + 1); 
    for (int i = 1; i <= n; ++i) {
        vertex[i] = i; // 假设顶点编号就是 1 到 n
    }

    // 2. 显式定义 edge 矩阵
    // 初始化邻接矩阵，默认所有点之间没有边 (INF)，自己到自己距离为 0
    vector<vector<int>> edge(n + 1, vector<int>(n + 1, INF));
    for (int i = 1; i <= n; ++i) {
        edge[i][i] = 0;
    }

    cout << "请依次输入 " << m << " 条边 (格式: 起点 终点 权重):" << endl;
    
    // 3. 读入 m 条边并建图（有向图）
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        
        // 【关键点：有向图只添加 u -> v 的单向边】
        // 使用 min 是为了防止输入数据中存在重边时，保留权重最小的一条边
        edge[u][v] = min(edge[u][v], w); 
    }

    // 4. 打印邻接矩阵
    cout << "\n--- 有向图的邻接矩阵 ---" << endl;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (edge[i][j] == INF) cout << "INF\t";
            else cout << edge[i][j] << "\t";
        }
        cout << endl;
    }

    return 0;
}
```

## 2. 邻接表版本 (Adjacency List)

```cpp
#include <iostream>
#include <vector>

using namespace std;

// 显式定义：边表节点（代表图中的一条边）
struct EdgeNode {
    int adjvex;          // 邻接点域：存储该边指向的顶点的下标/编号
    int weight;          // 权值域：存储边权（针对网图，非网图可省略）
    EdgeNode* next;      // 链域：指向同起点的下一条边

    // 构造函数，方便 new 的时候直接初始化
    EdgeNode(int v, int w = 0) : adjvex(v), weight(w), next(nullptr) {}
};

// 显式定义：顶点表节点
struct VertexNode {
    int data;            // 顶点域：存储顶点本身的有效信息（如字符、对象等）
    EdgeNode* firstEdge; // 边表头指针：指向该顶点的第一条出边

    // 构造函数
    VertexNode() : data(0), firstEdge(nullptr) {}
};

int main() {
    int n, m; 
    // n 表示顶点数，m 表示有向边数
    if (!(cin >> n >> m) || n == 0) return 0;

    // 1. 显式定义顶点数组 (这里数组下标从 1 到 n)
    vector<VertexNode> vertex(n + 1);

    // 2. 初始化顶点表
    for (int i = 1; i <= n; ++i) {
        vertex[i].data = i; // 这里简单地用编号作为顶点的数据
        vertex[i].firstEdge = nullptr; // 初始时每个顶点都没有出边
    }

    // 3. 处理 m 条有向边，显式构建单链表
    for (int i = 0; i < m; ++i) {
        int u, v; // 表示存在一条从 u 指向 v 的有向边
        cin >> u >> v;
        
        // 创建新的边节点
        EdgeNode* newEdge = new EdgeNode(v);
        
        // 采用“头插法”将新边插入到 u 顶点的出边链表中 (时间复杂度 O(1))
        // 注意：头插法会导致同一顶点的出边顺序与输入顺序相反，这在大多数图算法中并不影响结果
        newEdge->next = vertex[u].firstEdge;
        vertex[u].firstEdge = newEdge;
    }

    // 4. 打印邻接表，方便课堂画图演示
    cout << "--- 有向图的邻接表表示 ---" << endl;
    for (int i = 1; i <= n; ++i) {
        cout << "顶点 [" << vertex[i].data << "] ";
        
        // 遍历该顶点的边表（单链表遍历）
        EdgeNode* p = vertex[i].firstEdge;
        while (p != nullptr) {
            cout << "-> (指向:" << p->adjvex << ") ";
            p = p->next;
        }
        cout << "-> nullptr" << endl;
    }

    // 5. 释放动态分配的内存（遍历释放所有链表节点，防止内存泄漏）
    for (int i = 1; i <= n; ++i) {
        EdgeNode* p = vertex[i].firstEdge;
        while (p != nullptr) {
            EdgeNode* temp = p; // 暂存当前节点
            p = p->next;        // 移动到下一个节点
            delete temp;        // 释放暂存节点
        }
    }

    return 0;
}
```
